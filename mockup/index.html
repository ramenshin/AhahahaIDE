DAILY_QUOTA = 10000 COST_PER_SEARCH = 100 def __init__(self, api_key: str): self.api_key = api_key self.client = build("youtube", "v3", developerKey=api_key) self.quota_used = 0 def get_videos(self, channel_id: str, max_results: int = 50) -> List[dict]: # TODO: 레이트 리밋 처리 개선 필요 (Claude 리뷰 지적사항) if self.quota_used + self.COST_PER_SEARCH > self.DAILY_QUOTA: raise RuntimeError("일일 quota 초과") request = self.client.search().list( part="snippet", channelId=channel_id, maxResults=max_results, order="date", ) response = request.execute() self.quota_used += self.COST_PER_SEARCH return response.get("items", [])
